        const playersContainer = document.getElementById('players');
        players.forEach(player => {
            const playerElement = document.createElement('div');
            playerElement.classList.add('player', 'bg-gray-200', 'p-4', 'rounded-md', 'mb-2');
            playerElement.setAttribute('data-id', player.id);
            playerElement.textContent = `Jugador ${player.id + 1} - ${player.isImpostor ? (giveHint ? 'Impostor' : 'Impostor') : 'Civil'}`;

});
